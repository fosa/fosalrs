\subsection{Dirac-Delta Funktion}

\[ 
	\widetilde{\sigma}(t, \Delta t) = \left\{
		\begin{array}{c c}
			1 & t > \Delta t \\
			\frac{1}{2} + \frac{t}{2 \cdot \Delta t} &
				- \Delta t \leq t \leq \Delta t \\
			0 & t < - \Delta t
		\end{array}
		\right.
\]

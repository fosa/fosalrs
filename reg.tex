\subsection{Prüfung auf Linearität}
\begin{enumerate}
  \item Übertragungsprinzip erfüllt 
        \[ y = f(u) \qquad f(u_1 + u_2) = y_1 + y_2 \]
  \item Verstärkungsprinzip erfüllt
        \[ y = f(u) \qquad f(k \cdot u) = k \cdot y \]
\end{enumerate}

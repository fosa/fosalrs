\subsubsection{Überlagerungsprinzip}
\[
    y_1 = f(u_1)
\]
\[
    y_2 = f(u_2)
\]
\[
    f(u_1 + u_2) = y_1 + y_2
\]

\subsubsection{Verstärkungsprinzip}
\[
    y = f(u)
\]
\[
    f(k \cdot u) = k \cdot y
\]

\subsection{Zeitinvarianz}
\[
    y = f(u) \neq f(t)
\]
